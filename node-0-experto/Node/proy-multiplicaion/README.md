# Instalaciones
Para instalar todos los paquetes es necesario ejecutar el comando npm install

# Notas:
Opciones:
      --help     Muestra ayuda                                        [booleano]
      --version  Muestra número de versión                            [booleano]
  -b, --base     Base de la tabla de multiplicar            [número] [requerido]
  -l, --listar   Muestra la tabla en consola
                                         [booleano] [requerido] [defecto: false]
  -h, --hasta    hasta que valor multiplicar              [número] [defecto: 10]
